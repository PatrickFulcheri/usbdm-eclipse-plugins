88 
89  // Disable UART before changing registers
90  uart->C2 &= ~(UART_C2_TE_MASK | UART_C2_RE_MASK);
91 
92  // Calculate UART clock setting (5-bit fraction at right)
93  int scaledBaudValue = (2*clockFrequency)/baudrate;
94 
95 #ifdef UART_C4_BRFA_MASK
96  // Set Baud rate register
97  uart->BDH = (uart->BDH&~UART_BDH_SBR_MASK) | UART_BDH_SBR((scaledBaudValue>>(8+5)));
98  uart->BDL = UART_BDL_SBR(scaledBaudValue>>5);
99  // Fractional divider to get closer to the baud rate
100  uart->C4 = (uart->C4&~UART_C4_BRFA_MASK) | UART_C4_BRFA(scaledBaudValue);
101 #else
102  scaledBaudValue += 1<<4; // Round value
103  // Set Baud rate register
104  uart->BDH = (uart->BDH&~UART_BDH_SBR_MASK) | UART_BDH_SBR((scaledBaudValue>>(8+5)));
105  uart->BDL = UART_BDL_SBR(scaledBaudValue>>5);
106 #endif
107 
108  uart->C1 = 0;

111  // Enable UART Tx & Rx - with Rx IRQ
