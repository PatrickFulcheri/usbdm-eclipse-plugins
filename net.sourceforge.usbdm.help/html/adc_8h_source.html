399  /* ADC calibrate
400  *
401  * Calibrates the ADC
402  *
403  * @note Set up the ADC clock and resolution before calibration
404  *
405  * @return E_NO_ERROR Calibration successful
406  * @return E_CALIBRATE_FAIL Failed calibration
407  */
408  static ErrorCode calibrate() {
409 
410  // Save current SC3 as modified
411  uint8_t sc3 = adc->SC3;
412 
413  // Start calibration
414  setAveraging(AdcAveraging_Cal);
415 
416  // Wait for calibration to complete
417  while (adc->SC3 & ADC_SC3_CAL_MASK) {
418  __asm__("nop");
419  }
420 
421  bool failed = adc->SC3 & ADC_SC3_CALF_MASK;
422 
423  // Restore original SC3 value
424  adc->SC3 = sc3;
425 
426  // Check calibration outcome
427  if(failed) {
428  // Failed calibration
429  return setErrorCode(E_CALIBRATE_FAIL);
430  }
431 
432  // Calibration factor
433  uint16_t calib;
434 
435  calib = adc->CLPS + adc->CLP4 + adc->CLP3 + adc->CLP2 + adc->CLP1 + adc->CLP0;
436  calib /= 2;
437  calib |= (1<<15); // Set MSB
438  adc->PG = calib;
439 
440  calib = 0;
441  calib += adc->CLMS + adc->CLM4 + adc->CLM3 + adc->CLM2 + adc->CLM1 + adc->CLM0;
442  calib /= 2;
443  calib |= (1<<15); // Set MSB
444  adc->MG = calib;
445 
446  return E_NO_ERROR;
447  }
