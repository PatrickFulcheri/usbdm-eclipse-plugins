Detailed Description

template<int left, int right>
class USBDM::GpioCField< left, right >

Convenience template for GpioC fields.

See Field_T

Usage

using namespace USBDM;
// Instantiate for bit 6 down to 3 of GpioC
GpioCField<6,3> GpioC6_3;
// Set as digital output
GpioC6_3.setOutput();
// Write value to field
GpioC6_3.write(0x53);
// Clear all of field
GpioC6_3.bitClear();
// Clear lower two bits of field
GpioC6_3.bitClear(0x3);
// Set lower two bits of field
GpioC6_3.bitSet(0x3);
// Set as digital input
GpioC6_3.setInput();
// Read pin as int value
int x = GpioC6_3.read();
Template Parameters
leftBit number of leftmost bit in port (inclusive)
rightBit number of rightmost bit in port (inclusive)
