Detailed Description

template<uint32_t clockMask, uint32_t pcrAddress, uint32_t gpioAddress, int bitNum, Polarity polarity>
class USBDM::GpioBase_T< clockMask, pcrAddress, gpioAddress, bitNum, polarity >

Template representing a pin with Digital I/O capability.

Example

// Instantiate
using Pta3 = USBDM::Gpio_T<SIM_SCGC5_PORTA_MASK, PORTA_BasePtr, GPIOA_BasePtr, 3>;
// Set as digital output
Pta3::setOutput();
// Set pin high
Pta3::set();
// Set pin low
Pta3::clear();
// Toggle pin
Pta3::toggle();
// Set pin to boolean value
Pta3::write(true);
// Set pin to boolean value
Pta3::write(false);
// Set as digital input
Pta3::setInput();
// Read pin as boolean value
bool x = Pta3::read();
Template Parameters
clockMaskClock mask for PORT (PCR register) associated with GPIO
pcrAddressAddress of PORT (PCR register array) associated with GPIO
gpioAddressGPIO hardware address
bitNumBit number within PORT/GPIO
polarityEither USBDM::ActiveHigh or USBDM::ActiveLow
