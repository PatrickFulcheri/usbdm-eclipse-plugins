363  // limited to 0x7FFF for sensible PWM operation

401  *
402  * @param[in] period Period in seconds as a float
403  *
404  * @note Adjusts Timer pre-scaler to appropriate value.
405  * This will affect all channels of the timer.
406  *

408  * @return E_TOO_SMALL => failed to find suitable values
409  * @return E_TOO_LARGE => failed to find suitable values
410  */
411  static ErrorCode setPeriod(float period) {
412  float inputClock = Info::getInputClockFrequency();
413  int prescaleFactor=1;
414  int prescalerValue=0;
415 
416  // Maximum period value in ticks
417  uint32_t maxPeriodInTicks = 65536;
418 
419  // Check if CPWMS is set (affects period calculation)
420  if (tmr->SC&FTM_SC_CPWMS_MASK) {
421  // Centre-aligned period is ~double the MOD value but MOD is
422  // limited to 0x7FFF for sensible PWM operation so
423  // period in ticks is limited to 2*0x7FFF
424  maxPeriodInTicks = 65534;
425  }
426  while (prescalerValue<=7) {
427  float clock = inputClock/prescaleFactor;
428  uint32_t periodInTicks = round(period*clock);
429  if (periodInTicks < Info::minimumResolution) {
430  // Too short a period for 1% resolution
431  return setErrorCode(E_TOO_SMALL);
432  }
433  if (periodInTicks <= maxPeriodInTicks) {
434  tmr->SC = (tmr->SC&~FTM_SC_PS_MASK)|FTM_SC_PS(prescalerValue);
435  setPeriodInTicks(periodInTicks);
436  return E_NO_ERROR;
437  }
438  prescalerValue++;
439  prescaleFactor <<= 1;
440  }
441  // Too long a period
442  return setErrorCode(E_TOO_LARGE);
443  }
