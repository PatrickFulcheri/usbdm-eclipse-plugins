251  /**
252  * Enter Run Mode.
253  *
254  * This may be used to change between supported RUN modes (RUN, VLPR, HSRUN).\n
255  * Only the following transitions are allowed: VLPR <-> RUN <-> HSRUN.
256  *
257  * @param[in] smcRunMode Mode to enter
258  *
259  * @return E_NO_ERROR No error
260  * @return E_ILLEGAL_POWER_TRANSITION Cannot transition to smcRunMode from current run mode
261  */
262  static ErrorCode enterRunMode(SmcRunMode smcRunMode) {
263 #ifdef SMC_PMPROT_AHSRUN
264  SmcStatus smcStatus = getPowerStatus();
265 #endif
266  switch(smcRunMode) {
267  case SmcRunMode_Normal:
268  smc->PMCTRL = (smc->PMCTRL&~SMC_PMCTRL_RUNM_MASK)|smcRunMode;
269  while (getPowerStatus() != SmcStatus_run) {
270  __asm__("nop");
271  }
272  break;
273 #ifdef SMC_PMPROT_AHSRUN
274  case SmcRunMode_HighSpeed:
275  if (smcStatus != SmcStatus_run) {
276  // Can only transition from RUN mode
277  return setErrorCode(E_ILLEGAL_POWER_TRANSITION);
278  }
279  smc->PMCTRL = (smc->PMCTRL&~SMC_PMCTRL_RUNM_MASK)|smcRunMode;
280  while (getPowerStatus() != SmcStatus_hsrun) {
281  __asm__("nop");
282  }
283  break;
284 #endif
285  case SmcRunMode_VeryLowPower:
286 #ifdef SMC_PMPROT_AHSRUN
287  if (smcStatus != SmcStatus_run) {
288  // Can only transition from RUN mode
289  return setErrorCode(E_ILLEGAL_POWER_TRANSITION);
290  }
291 #endif
292  smc->PMCTRL = (smc->PMCTRL&~SMC_PMCTRL_RUNM_MASK)|smcRunMode;
293  while (getPowerStatus() != SmcStatus_vlpr) {
294  __asm__("nop");
295  }
296  break;
297  default:
298  return setErrorCode(E_ILLEGAL_PARAM);
299  }
300  return E_NO_ERROR;
301  }

352  /**
353  * Enter Wait Mode (WAIT, VLPW)\n
354  * (ARM core SLEEP mode)
355  *
356  * The processor will stop execution and enter SLEEP mode.\n
357  * In this mode the core clock is disabled (no code executing),
358  * but bus clocks are enabled (peripheral modules are operational).
359  *
360  * This function can be used to enter normal WAIT mode or VLPW mode
361  * depending upon current run mode.
362  *
363  * Mode transitions:
364  * - RUN -> WAIT
365  * - VLPR -> VLPW
366  *
367  * WAIT mode is exited using any enabled interrupt or RESET.
368  *
369  * For Kinetis K:
370  * If in VLPW mode, the statue of the SMC_PMCTRL[LPWUI] bit
371  * determines if the processor exits to VLPR or RUN mode.\n
372  * Use setExitVeryLowPowerOnInterrupt() to modify this action.
373  *
374  * For Kinetis L:
375  * LPWUI does not exist.\n
376  * Exits with an interrupt from VLPW will always be back to VLPR.\n
377  * Exits from an interrupt from WAIT will always be back to RUN.
378  *
379  * @note Some modules include a programmable option to disable them in wait mode.\n
380  * If those modules are programmed to disable in wait mode, they will not be able to
381  * generate interrupts to wake the core.
382  */
383  static void enterWaitMode() {
384  SCB->SCR &= ~SCB_SCR_SLEEPDEEP_Msk;
385  // Make sure write completes
386  __DSB();
387  __WFI();
388  }
