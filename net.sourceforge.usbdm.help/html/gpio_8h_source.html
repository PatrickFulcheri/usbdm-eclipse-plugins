686  PinPull pinPull,

688  PinFilter pinFilter = PinFilter_None
689  ) {
690  setInput(pinPull|pinIrq|pinFilter);
691  }

693  * Set individual pin directions
694  *
695  * @param[in] mask Mask for pin directions (1=>out, 0=>in)
696  */
697  static __attribute__((always_inline)) void setDirection(uint32_t mask) {
698  gpio->PDDR = (gpio->PDDR&~MASK)|((mask<<right)&MASK);
699  }

701  * Set bits in field
702  *
703  * @param[in] mask Mask to apply to the field (1 => set bit, 0 => unchanged)
704  */
705  static __attribute__((always_inline)) void bitSet(const uint32_t mask) {
706  gpio->PSOR = (mask<<right)&MASK;
707  }
708  /**
709  * Clear bits in field
710  *
711  * @param[in] mask Mask to apply to the field (1 => clear bit, 0 => unchanged)
712  */
713  static __attribute__((always_inline)) void bitClear(const uint32_t mask) {
714  gpio->PCOR = (mask<<right)&MASK;
715  }
716  /**
717  * Toggle bits in field
718  *
719  * @param[in] mask Mask to apply to the field (1 => toggle bit, 0 => unchanged)
720  */
721  static __attribute__((always_inline)) void bitToggle(const uint32_t mask) {
722  gpio->PTOR = (mask<<right)&MASK;
723  }
724  /**
725  * Read field
726  *
727  * @return value from field
728  */
729  static __attribute__((always_inline)) uint32_t read() {
730  return ((gpio->PDIR) & MASK)>>right;
731  }
732  /**
733  * Write field
734  *
735  * @param[in] value to insert as field
736  */
737  static __attribute__((always_inline)) void write(uint32_t value) {
