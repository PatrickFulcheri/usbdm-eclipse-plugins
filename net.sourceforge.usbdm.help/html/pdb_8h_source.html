220  /**
221  * Get 'best' dividers for given period.
222  * This involves finding the smallest prescaler that allows the PDB period to be set to greater
223  * than the given period.\n
224  * This produces the highest resolution.\n
225  * It is quite possible that other values would be more suitable for a particular application.
226  * For example, carefully chosen prescalers may result in less rounding for the needed intermediate
227  * points for pulse outputs etc.
228  *
229  * @param[in] period Period in seconds as a float
230  * @param[out] multValue Determined pdb_sc_mult value
231  * @param[out] prescaleValue Determined pdb_sc_prescaler value
232  *
233  * @return E_NO_ERROR => success
234  * @return E_ERROR => failed to find suitable values
235  */
236  static ErrorCode getDividers(float period, uint32_t &multValue, int &prescaleValue) {
237 
238  // Multiplier factors for prescale divider
239  static const int multFactors[] = {1,10,20,40};
240 
241  float inputClock = Info::getInputClockFrequency();
242 
243  // No MOD value found so far
244  uint32_t mod = 0;
245 
246  // Try each divider multiplier
247  for (unsigned trialMultValue=0; trialMultValue<(sizeof(multFactors)/sizeof(multFactors[0])); trialMultValue++) {
248  int multfactor = multFactors[trialMultValue];
249 
250  // Try prescalers from smallest to largest
251  // Find first prescaler for which a suitable modulo exists
252  int prescaleFactor=1;
253  for (unsigned trialPrescaleValue=0; trialPrescaleValue<=7; trialPrescaleValue++) {
254  float clock = inputClock/(multfactor*prescaleFactor);
255  uint32_t trialMod = round(period*clock)-1;
256  // printf("multfactor=%2d, prescaleFactor = %3d, mod=%8d, period=%f\n", multfactor, prescaleFactor, trialMod, period);
257  if (trialMod <= 0) {
258  // Too short a period
259  return E_TOO_SMALL;
260  }
261  if (trialMod <= 65535) {
262  if (trialMod>mod) {
263  // Better value - save
264  prescaleValue = trialPrescaleValue;
265  multValue = trialMultValue;
266  mod = trialMod;
267  }
268  break;
269  }
270  prescaleFactor <<= 1;
271  }
272  }
273  return setErrorCode((mod==0)?E_ERROR:E_NO_ERROR);
274  }
