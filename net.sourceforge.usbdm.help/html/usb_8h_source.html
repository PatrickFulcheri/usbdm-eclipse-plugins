704  * Handles SETUP Packet
705  */
706 template<class Info, int EP0_SIZE>
707 void UsbBase_T<Info, EP0_SIZE>::handleSetupToken() {
708 
709  // Save data from SETUP packet
710  memcpy(&ep0SetupBuffer, controlEndpoint.getBuffer(), sizeof(ep0SetupBuffer));
711 
712  controlEndpoint.setupReceived();
713 
714  // Call-backs only persist during a SETUP transaction
715  setSetupCompleteCallback(nullptr);
716 
717 // PRINTF("handleSetupToken(%s)\n", reportSetupPacket(&ep0SetupBuffer));
718 
719  switch(REQ_TYPE(ep0SetupBuffer.bmRequestType)) {
720  case REQ_TYPE_STANDARD :
721  // Standard device requests
722  switch (ep0SetupBuffer.bRequest) {
723  case GET_STATUS : handleGetStatus(); break;
724  case CLEAR_FEATURE : handleClearFeature(); break;
725  case SET_FEATURE : handleSetFeature(); break;
726  case SET_ADDRESS : handleSetAddress(); break;
727  case GET_DESCRIPTOR : handleGetDescriptor(); break;
728  case GET_CONFIGURATION : handleGetConfiguration(); break;
729  case SET_CONFIGURATION : handleSetConfiguration(); break;
730  case SET_INTERFACE : handleSetInterface(); break;
731  case GET_INTERFACE : handleGetInterface(); break;
732  case SET_DESCRIPTOR :
733  case SYNCH_FRAME :
734  default : handleUnexpected(); break;
735  }
736  break;
737 
738  case REQ_TYPE_CLASS :
739  // Class requests
740  handleUnexpected();
741  break;
742 
743  case REQ_TYPE_VENDOR :
744  // Handle special commands here
745  switch (ep0SetupBuffer.bRequest) {
746 #ifdef MS_COMPATIBLE_ID_FEATURE
747  case MS_VENDOR_CODE :
748  // PUTS("REQ_TYPE_VENDOR - VENDOR_CODE");
749  if (ep0SetupBuffer.wIndex == 0x0004) {
750  // PUTS("REQ_TYPE_VENDOR - MS_CompatibleIdFeatureDescriptor");
751  ep0StartTxTransaction(sizeof(MS_CompatibleIdFeatureDescriptor), (const uint8_t *)&msCompatibleIdFeatureDescriptor);
752  }
753  else if (ep0SetupBuffer.wIndex == 0x0005) {
754  // PUTS("REQ_TYPE_VENDOR - MS_PropertiesFeatureDescriptor");
755  ep0StartTxTransaction(sizeof(MS_PropertiesFeatureDescriptor), (const uint8_t *)&msPropertiesFeatureDescriptor);
756  }
757  else {
758  handleUnexpected();
759  }
760  break;
761 #endif
762 
763  default :
764  handleUnexpected();
765  break;
766  }
767  break;
768 
769  case REQ_TYPE_OTHER :
770  handleUnexpected();
771  break;
772  }
773  // In case another SETUP packet
774  controlEndpoint.initialiseBdtRx();
775 
776  // Allow transactions post SETUP packet (clear TXSUSPENDTOKENBUSY)
777  usb->CTL = USB_CTL_USBENSOFEN_MASK;
778 }
