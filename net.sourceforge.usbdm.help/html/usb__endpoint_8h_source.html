412  /**
413  * Handle OUT [Receive, device <- host, DATA0/1]
414  */
415  void handleOutToken() {
416  uint8_t transferSize = 0;
417 // pushState('O');
418 
419  // Toggle DATA0/1 for next pkt
420  rxData1 = !rxData1;
421 
422  switch (state) {
423  case EPDataOut: // Receiving a sequence of OUT packets
424  // Save the data from the Receive buffer
425  transferSize = saveRxData();
426 // if (fEndpointNumber == 1) {
427 // PRINTF("ep1.handleOutToken() s=%d. size=%d, r=%d\n", state, transferSize, fDataRemaining);
428 // }
429  // Complete transfer on undersize packet or received expected number of bytes
430  if ((transferSize < EP_MAXSIZE) || (fDataRemaining == 0)) {
431  // Now idle
432  state = EPIdle;
433  doCallback(EPDataOut);
434  }
435  else {
436  // Set up for next OUT packet
437  initialiseBdtRx();
438  }
439  break;
440 
441  case EPStatusOut: // Done OUT packet as a status handshake from host (IN CONTROL transfer)
442  // No action
443  state = EPIdle;
444  doCallback(EPStatusOut);
445  break;
446 
447  // We don't expect an OUT token while in the following states
448  case EPDataIn: // Doing a sequence of IN packets (until data count <= EP_MAXSIZE)
449  case EPStatusIn: // Just done an IN packet as a status handshake
450  case EPIdle: // Idle
451  case EPComplete: // Not used
452  case EPStall: // Not used
453  case EPThrottle: // Not used
454  PRINTF("Unexpected OUT, s = %d\n", state);
455  state = EPIdle;
456  break;
457  }
458  }
459  /**
460  * Handle IN token [Transmit, device -> host]
461  */
462  void handleInToken() {
463  // Toggle DATA0/1 for next packet
464  txData1 = !txData1;
465 
466  // PUTS(fHardwareState[BDM_OUT_ENDPOINT].data0_1?"ep2HandleInToken-T-1\n":"ep2HandleInToken-T-0\n");
467 
468  switch (state) {
469  case EPDataIn: // Doing a sequence of IN packets
470  // Check if packets remaining
471  if ((fDataRemaining > 0) || fNeedZLP) {
472  // Set up next IN packet
473  initialiseBdtTx();
474  }
475  else {
476  state = EPIdle;
477  // Execute callback function to process previous OUT data
478  doCallback(EPDataIn);
479  }
480  break;
481 
482 
483  case EPStatusIn: // Just done an IN packet as a status handshake for an OUT Data transfer
484  // Now Idle
485  state = EPIdle;
486  // Execute callback function to process previous OUT data
487  doCallback(EPStatusIn);
488  break;
489 
490  // We don't expect an IN token while in the following states
491  case EPIdle: // Idle (Transmit complete)
492  case EPDataOut: // Doing a sequence of OUT packets (until data count <= EP_MAXSIZE)
493  case EPStatusOut: // Doing an OUT packet as a status handshake
494  default:
495  PRINTF("Unexpected IN, %d\n", state);
496  state = EPIdle;
497  break;
498  }
